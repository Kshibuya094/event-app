function isWithinAllowedTime() { const now = new Date(); const jstNow = new Date(now.getTime() + (9 * 60 * 60 * 1000)); // JST補正 const start = new Date("2025-07-30T22:25:00+09:00"); const end = new Date("2025-07-30T22:35:00+09:00"); return jstNow >= start && jstNow <= end; } function capture() { if (!isWithinAllowedTime()) { alert("現在はQRコードの読み取り時間外です。\n有効時間：2025年7月30日 22:25〜22:35"); return; } // 撮影処理続行 }
