// 有効時間（例：2025年7月30日 22:40〜22:43） const startTimeJST = new Date("2025-07-30T22:40:00+09:00").getTime(); const endTimeJST = new Date("2025-07-30T22:43:00+09:00").getTime(); function isWithinAllowedTime() { const now = Date.now(); return now >= startTimeJST && now <= endTimeJST; } // 撮影ボタンなどで使用 function capture() { if (!isWithinAllowedTime()) { alert("現在はQRコードの読み取り時間外です。"); return; } // 撮影処理続行 }
