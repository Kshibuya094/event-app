function getJSTTime() { const now = new Date(); const jstOffset = 9 * 60 * 60 * 1000; // JSTはUTC+9 return now.getTime() + jstOffset - now.getTimezoneOffset() * 60 * 1000; } function isWithinAllowedTime() { const nowJST = getJSTTime(); const start = new Date("2025-07-30T22:25:00+09:00").getTime(); const end = new Date("2025-07-30T22:35:00+09:00").getTime(); return nowJST >= start && nowJST <= end; }
